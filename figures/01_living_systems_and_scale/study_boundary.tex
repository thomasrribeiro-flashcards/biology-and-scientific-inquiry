% flashcards-desc: A potted plant sits in soil with surrounding air. A small dashed outline labeled A encloses only the plant. A larger dashed outline labeled B encloses the plant, the soil, and nearby air. Arrows link soil to plant and air to plant to show interactions.

\begin{tikzpicture}[fc text]

  % --- components ---
  % pot with soil
  \fill[fcsoil] (2.1,0.55) -- (3.9,0.55) -- (3.65,1.55) -- (2.35,1.55) -- cycle;
  \draw[fcink, line width=0.9pt] (2.1,0.55) -- (3.9,0.55) -- (3.65,1.55) -- (2.35,1.55) -- cycle;
  \node[fc note, text=white] at (3.0,1.05) {soil};

  % plant: stem and two leaves
  \draw[fcgreen, line width=1.6pt] (3.0,1.55) -- (3.0,3.6);
  \draw[fcgreen, line width=1.3pt, fill=fcgreen!25, rotate around={35:(2.62,3.12)}]
      (2.62,3.12) ellipse (0.42 and 0.19);
  \draw[fcgreen, line width=1.3pt, fill=fcgreen!25, rotate around={-35:(3.38,2.82)}]
      (3.38,2.82) ellipse (0.42 and 0.19);
  \node[fc label, text=fcgreen] at (4.7,3.0) {plant};

  % air region marker
  \node[fc note] at (1.15,3.7) {air};
  \node[circle, draw=fcblue, line width=0.7pt, minimum size=3mm] at (1.5,3.2) {};
  \node[circle, draw=fcblue, line width=0.7pt, minimum size=2mm] at (0.95,2.9) {};
  \node[circle, draw=fcblue, line width=0.7pt, minimum size=2.5mm] at (1.7,4.15) {};

  % --- interaction arrows (generic: one component affecting another) ---
  \draw[fc arrow] (2.15,2.6) -- (1.25,2.9); % plant <-> air region
  \draw[fc arrow] (3.0,1.75) -- (3.0,1.05);  % soil <-> plant (into soil)

  % --- candidate boundary A: plant only ---
  \draw[fc boundary] (2.05,1.9) rectangle (3.95,3.95);
  \node[fc label, text=fcorange] at (3.0,4.2) {boundary A};

  % --- candidate boundary B: plant + soil + air ---
  \draw[fc boundary] (0.35,0.2) rectangle (5.7,4.95);
  \node[fc label, text=fcorange, fill=fcpale, inner sep=1.5pt] at (4.55,0.5) {boundary B};
